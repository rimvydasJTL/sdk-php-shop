# (M14) Create a thread with a seller

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\Core\Domain\FileWrapper;
use Mirakl\MMP\Common\Domain\Inbox\CreateInboxThreads\Entity;
use Mirakl\MMP\Common\Domain\Inbox\CreateInboxThreads\FreeTextTopic;
use Mirakl\MMP\Common\Domain\Inbox\CreateInboxThreads\ThreadInput;
use Mirakl\MMP\Shop\Request\Inbox\CreateInboxThreadsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$entity = new Entity();
$entity->setType('ORDER');

$topic = new FreeTextTopic();
$topic->setValue('Question about shipping delay');

$threadInput = new ThreadInput();
$threadInput->setBody('Hello, I have a question about the shipping delay on my recent order.')
    ->setEntity($entity)
    ->setTopic($topic);

$request = new CreateInboxThreadsRequest();
$request->setThreadInput($threadInput);

// Optionally attach files
$file = new FileWrapper(new \SplFileObject('/path/to/attachment.pdf'));
$request->addFile($file);

$result = $api->createInboxThreads($request);
// $result => @see \Mirakl\MMP\Common\Domain\Inbox\InboxThreadsResponse
// $result->getThreadId()
// $result->getMessageId()
```
