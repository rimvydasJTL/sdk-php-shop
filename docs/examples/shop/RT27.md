# (RT27) Mark a return as closed

```php
use Mirakl\MMP\Common\Domain\Returns\CloseReturns\Returns;
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Returns\CloseReturnsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$return = new Returns();
$return->setId('return-001');

$request = new CloseReturnsRequest();
$request->setReturns([$return]);

$result = $api->closeReturns($request);
// $result => @see \Mirakl\MMP\Common\Domain\Returns\Returns

$errors = $result->getReturnErrors();
$successes = $result->getReturnSuccess();
```
