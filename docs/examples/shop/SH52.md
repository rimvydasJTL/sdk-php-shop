# (SH52) Get shipping charges configured for a shop account

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Shipping\GetShippingChargesShopRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$request = new GetShippingChargesShopRequest();
$request->setOverriddenByShop(true);

$result = $api->getShippingChargesShop($request);
// $result => @see \Mirakl\MMP\Shop\Domain\Shipping\ShippingChargesShopResponse

$result->getShopId();
$result->getCurrencyIsoCode();
$result->getLastUpdatedDate();

foreach ($result->getConfiguration() as $config) {
    $config->getLogisticClassCode();
    $config->getShippingMethodCode();
    $config->getShippingZoneCode();
    $config->getOverriddenByShop();

    $prices = $config->getPrices();
    $prices->getType();              // FIXED_PRICE, ORDER_RANGE, FREE_SHIPPING, ...
    $prices->getInitialPrice();
    $prices->getAdditionalPrice();
    $prices->getFreeShippingThreshold();

    foreach ($prices->getRanges() as $range) {
        $range->getThreshold();
        $range->getPrice();
    }
}
```
