# (RT01) Create a return request

```php
use Mirakl\MMP\Common\Domain\Collection\Returns\CreateReturnCollection;
use Mirakl\MMP\Common\Domain\Returns\CreateReturn;
use Mirakl\MMP\Common\Domain\Returns\ReturnLine;
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Returns\CreateReturnRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$returnLine = new ReturnLine();
$returnLine->setOrderLineId('ORDER-001-A-1');
$returnLine->setQuantity(1);

$createReturn = new CreateReturn();
$createReturn->setOrderCommercialId('ORDER-001');
$createReturn->setReasonCode('RETURN_REASON_DEFECTIVE');
$createReturn->setReturnLines([$returnLine]);

$returns = new CreateReturnCollection();
$returns->add($createReturn);

$request = new CreateReturnRequest($returns);
$result = $api->createReturn($request);

// $result => @see \Mirakl\MMP\Common\Domain\Returns\Returns
// $result->getReturnSuccess() => successfully created returns
// $result->getReturnErrors()  => returns that failed to be created
```
