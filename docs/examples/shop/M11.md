# (M11) List all threads

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Inbox\GetInboxThreadsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');
$request = new GetInboxThreadsRequest();
// Optional parameters
$request->setChannelCodes('CHANNEL_CODES')
    ->setEntityId('ENTITY_ID')
    ->setEntityType('ENTITY_TYPE')
    ->setUpdatedSince('UPDATED_SINCE')
    ->setWithMessages('WITH_MESSAGES');
$result = $api->getInboxThreads($request);
```
