# (ST31) Update shipment additional information

```php
use Mirakl\MMP\Common\Domain\Shipment\Shipment;
use Mirakl\MMP\Common\Domain\Shipment\ShipmentAdditionalInformation;
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Shipment\UpdateShipmentsAdditionalInformationRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$stringInfo = new ShipmentAdditionalInformation();
$stringInfo->setCode('CARRIER_REFERENCE');
$stringInfo->setType('STRING');
$stringInfo->setValue('CR-2024-00456');

$dateInfo = new ShipmentAdditionalInformation();
$dateInfo->setCode('ESTIMATED_DELIVERY');
$dateInfo->setType('DATE');
$dateInfo->setValue('2024-07-01');

$shipment = new Shipment();
$shipment->setId('shipment-001');
$shipment->setShipmentAdditionalInformation([$stringInfo, $dateInfo]);

$request = new UpdateShipmentsAdditionalInformationRequest();
$request->setShipments([$shipment]);

$result = $api->updateShipmentsAdditionalInformation($request);
// $result => @see \Mirakl\MMP\Common\Domain\Shipment\ShipmentsAdditionalInformation

$errors = $result->getShipmentUpdateErrors();
$successes = $result->getShipmentUpdateSuccess();
```
