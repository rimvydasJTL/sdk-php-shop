# (A21) Get shop statistics

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Account\GetAccountStatisticsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$request = new GetAccountStatisticsRequest();
$request->setChannelCode('CHANNEL_WEB');
$request->setPeriods(['ALL_SINCE_BEGINNING', 'LAST_MONTH']);

$result = $api->getAccountStatistics($request);
// $result => @see \Mirakl\MMP\Shop\Domain\Account\AccountStatisticsResponse

$approvalRate = $result->getApprovalRate();
$ordersCount = $result->getOrdersCount();
$grade = $result->getGrade();
```
