# (ST07) Update shipment shipping origin

```php
use Mirakl\MMP\Common\Domain\Shipment\Shipment;
use Mirakl\MMP\Common\Domain\Shipment\ShippingFrom;
use Mirakl\MMP\Common\Domain\Shipment\ShippingFromWarehouse;
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Shipment\UpdateShipmentsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$warehouse = new ShippingFromWarehouse();
$warehouse->setCode('WAREHOUSE_EU_01');

$shippingFrom = new ShippingFrom();
$shippingFrom->setWarehouse($warehouse);

$shipment = new Shipment();
$shipment->setId('shipment-001');
$shipment->setShippingFrom($shippingFrom);

$request = new UpdateShipmentsRequest();
$request->setShipments([$shipment]);

$result = $api->updateShipments($request);
// $result => @see \Mirakl\MMP\Common\Domain\Shipment\Shipments

$errors = $result->getShipmentErrors();
$successes = $result->getShipmentSuccess();
```
