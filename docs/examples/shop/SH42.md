# (SH42) Get shop shipping configuration

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Shipping\GetShippingConfigurationShopRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');

$request = new GetShippingConfigurationShopRequest();
$result = $api->getShippingConfigurationShop($request);
// $result => @see \Mirakl\MMP\Shop\Domain\Shipping\ShippingConfigurationShopResponse

$result->getShopId();
$result->getOrderFreeShipping();
$result->getShippingPaidByOperator();
$result->getLastUpdatedDate();

$result->getLeadTimeToShip()->getDefaultValue();
$result->getLeadTimeToShip()->getShopValue();

foreach ($result->getDeliveryTimes() as $deliveryTime) {
    $deliveryTime->getEarliestDays();
    $deliveryTime->getLatestDays();
}

foreach ($result->getOrderCutOffTimes() as $cutOff) {
    $cutOff->getCutOffTime();
    $cutOff->getShippingTypeCode();
    $cutOff->getSetBy();
    $cutOff->isRequired();
}
```
