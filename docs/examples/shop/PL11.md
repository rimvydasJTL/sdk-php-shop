# (PL11) List picklists

```php
use Mirakl\MMP\Shop\Client\ShopApiClient;
use Mirakl\MMP\Shop\Request\Picklist\GetPicklistsRequest;

$api = new ShopApiClient('API_URL', 'API_KEY', 'SHOP_ID');
$request = new GetPicklistsRequest();
// Optional parameters
$request->setOrderLineId('ORDER_LINE_ID')
    ->setPicklistId('PICKLIST_ID')
    ->setPicklistState('PICKLIST_STATE')
    ->setPickrunCode('PICKRUN_CODE')
    ->setPickupDateMax('PICKUP_DATE_MAX');
$result = $api->getPicklists($request);
```
